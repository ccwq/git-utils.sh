# 可用环境变量:
# %APP_HOME%   -> 项目根目录绝对路径
# %APP_SH%     -> sh 目录绝对路径
# %APP_CONFIG% -> config 目录绝对路径

# 可以使用windows系统内置的环境变量
# %USERPROFILE% -> 用户目录绝对路径

# 系统
--update cd /d %APP_HOME% && git pull
-u wsha --update
--open code %APP_HOME%
--lazygit cd /d %APP_HOME% && lazygit
-lgit wsha --lazygit

# 编辑配置文件
--edit-config code %APP_CONFIG%/wsh-alias.txt
-ec wsha --edit-config

# 编辑用户配置文件
--edit-config-user code %USERPROFILE%/.config/wsh-alias.txt
-ecu wsha --edit-config-user

# docker
dc docker compose
