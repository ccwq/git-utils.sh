#!/bin/bash
# wsha-py.sh — 测试包装脚本，将测试框架的调用转发给 Python 版本的 wsha
# 处理 Git Bash 路径到 Windows 路径的转换，供原生 Windows Python 使用

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
PROJECT_ROOT="$(cd "$SCRIPT_DIR/.." && pwd)"

# 设置应用环境变量（与 sh/wsha.sh set_app_env 等价）
export APP_HOME="$PROJECT_ROOT"
export APP_SH="$PROJECT_ROOT/sh"
export APP_CONFIG="$PROJECT_ROOT/config"

# 将 Python 源码目录加入 PYTHONPATH
export PYTHONPATH="$PROJECT_ROOT/py${PYTHONPATH:+:$PYTHONPATH}"

# 辅助函数: 转换 Git Bash 路径为 Windows 路径（供 Windows Python 使用）
to_win_path() {
    if command -v cygpath &>/dev/null; then
        cygpath -w "$1" 2>/dev/null || echo "$1"
    else
        echo "$1"
    fi
}

to_unix_path() {
    if command -v cygpath &>/dev/null; then
        cygpath -u "$1" 2>/dev/null || echo "$1"
    else
        echo "$1"
    fi
}

# 转换 WSHA_CONFIG_FILE（Python 读文件用 Windows 路径）
if [[ -n "${WSHA_CONFIG_FILE:-}" ]]; then
    export WSHA_CONFIG_FILE_DISPLAY="$(to_unix_path "$WSHA_CONFIG_FILE")"
    export WSHA_CONFIG_FILE="$(to_win_path "$WSHA_CONFIG_FILE")"
fi

# 转换 HOME 路径（Python Path.home() 在 Windows 不识别 Git Bash 路径）
if [[ -n "${HOME:-}" ]]; then
    export WSHA_OVERRIDE_HOME="$(to_win_path "$HOME")"
fi

exec python3 -m wsha.cli "$@"
```

def get_default_config_paths() -> Dict[str, str]:
    """
    Get default config file paths with priority.
    Returns ordered dict of source_name -> config_path (builtin < user < project)
    """
    # Honor HOME/WSHA_OVERRIDE_HOME env vars (set by wsha-py.sh for tests)
    home_override = os.environ.get('WSHA_OVERRIDE_HOME') or os.environ.get('HOME')
    if home_override:
        home = Path(home_override)
    else:
        home = Path.home()

    app_home = os.environ.get('APP_HOME', '')
    configs = {}

    # Built-in: APP_HOME/config/wsh-alias.txt
    if app_home:
        builtin_path = Path(app_home) / "config" / "wsh-alias.txt"
        if builtin_path.exists():
            configs['builtin'] = str(builtin_path)

    # User-level: $HOME/.config/wsh-alias.txt  (note: .config not config)
    user_path = home / ".config" / "wsh-alias.txt"
    if user_path.exists():
        configs['user'] = str(user_path)

    # Project-level: $PWD/.config/wsh-alias.txt  (note: .config not config)
    local_path = Path.cwd() / ".config" / "wsh-alias.txt"
    if local_path.exists():
        configs['project'] = str(local_path)

    return configs
```

**修复 invoke_cmd 完整版本**:

```python
def invoke_cmd(cmd_text: str) -> int:
    """
    Execute the expanded command and return exit code.

    Per D-22, D-23:
    1. Expand %VAR% environment variables
    2. Detect complex shell commands (pipes, redirects, etc.)
    3. Execute appropriately (subprocess vs shell)
    4. Return standardized exit codes (0, 1, 127)

    Args:
        cmd_text: Command string to execute

    Returns:
        Exit code: 0 (success), 1 (error), 127 (command not found)
    """
    # Step 1: Expand environment variables
    cmd_text = expand_env_vars(cmd_text)

    # Step 2: Check if it's a complex shell command
    if not is_complex_shell_command(cmd_text):
        # Simple command - execute directly with subprocess
        # Note: NO capture_output so stdout/stderr pass through to terminal
        try:
            cmd_tokens = shlex.split(cmd_text)
            result = subprocess.run(cmd_tokens)
            return result.returncode
        except FileNotFoundError:
            import sys
            print(f"bash: {shlex.split(cmd_text)[0] if cmd_text else 'cmd'}: command not found", file=sys.stderr)
            return 127
        except Exception as e:
            import sys
            print(f"Error executing command: {e}", file=sys.stderr)
            return 1
    else:
        # Complex shell command - use bash -c
        # Note: NO capture_output so stdout/stderr pass through to terminal
        try:
            result = subprocess.run(["bash", "-c", cmd_text])
            return result.returncode
        except FileNotFoundError:
            import sys
            print("bash: command not found", file=sys.stderr)
            return 127
        except Exception as e:
            import sys
            print(f"Error executing command: {e}", file=sys.stderr)
            return 1
```
