```python
"""CLI entry point for wsha - alias command launcher."""

import os
import sys
import subprocess as sp
from typing import Tuple, Optional

import click

from .config import load_config, AliasEntry
from .expand import expand_template, invoke_cmd, print_alias_hit
from .matching import get_tokens
from .matcher import AliasMatcher
from .errors import ConfigParseError


SOURCE_LABEL_MAP = {
    'custom':  '[自定义]',
    'builtin': '[内置]',
    'user':    '[用户级]',
    'project': '[项目级]',
}


def to_display_path(path: str) -> str:
    """Convert path to Unix/Git Bash format for display output."""
    try:
        result = sp.run(['cygpath', '-u', path], capture_output=True, text=True)
        if result.returncode == 0:
            return result.stdout.strip()
    except FileNotFoundError:
        pass
    return path.replace('\\', '/')


def get_source_label(source_name: str) -> str:
    """Map internal source name to Chinese display label."""
    return SOURCE_LABEL_MAP.get(source_name, f'[{source_name}]')


def print_list(aliases: list, sources: dict) -> None:
    """
    Print alias list in table format matching shell version output.
    
    Format:
      别名                   命令
      [标签] /path/to/config
        alias_name           template
    """
    click.echo(f"{'别名':<20} {'命令'}")
    click.echo("")
    
    # Group by source
    by_source: dict = {}
    source_order: list = []
    for alias in aliases:
        sname = alias.source_name
        if sname not in by_source:
            by_source[sname] = []
            source_order.append(sname)
        by_source[sname].append(alias)
    
    for sname in source_order:
        source_path = sources.get(sname, '')
        display_path = to_display_path(source_path)
        # Check for WSHA_CONFIG_FILE_DISPLAY override (set by wsha-py.sh)
        if sname == 'custom' and os.environ.get('WSHA_CONFIG_FILE_DISPLAY'):
            display_path = os.environ['WSHA_CONFIG_FILE_DISPLAY']
        label = get_source_label(sname)
        click.echo(f"{label} {display_path}")
        for alias in by_source[sname]:
            click.echo(f"  {alias.name:<20} {alias.template}")
        click.echo("")


@click.command(context_settings=dict(ignore_unknown_options=True, allow_extra_args=True))
@click.argument('alias_input', required=False)
@click.argument('args', nargs=-1, type=click.UNPROCESSED)
@click.option('--list', '-l', 'do_list', is_flag=True, help='List all aliases in table format')
@click.option('--list-view', '-lv', 'do_list_view', is_flag=True, help='List all aliases (view mode)')
@click.option('--find', '-f', 'find_pattern', default=None, help='Search aliases by pattern')
@click.option('--cache-clear', is_flag=True, help='Clear the config cache')
@click.option('--help', '-h', 'show_help', is_flag=True, help='Show help')
def main(alias_input, args, do_list, do_list_view, find_pattern, cache_clear, show_help):
    """wsha - alias command launcher (Python implementation)"""
    
    if show_help or (not alias_input and not do_list and not do_list_view
                     and find_pattern is None and not cache_clear):
        click.echo("wsha - alias command launcher")
        click.echo("Usage: w <alias> [args...]  |  w --list  |  w --list-view")
        return

    # Handle --cache-clear (CFG-04)
    if cache_clear:
        from .cache import CacheManager
        CacheManager().clear()
        click.echo("Cache cleared.")
        return

    # Load config — errors are always printed to stderr
    aliases, errors, sources = load_config()
    
    if errors:
        for err in errors:
            click.echo(str(err), err=True)
        sys.exit(1)

    # Handle --list / -l (CLI-01)
    if do_list:
        print_list(aliases, sources)
        return

    # Handle --list-view / -lv (CLI-02)
    # In test env (WSHA_TEST_GRID_CAPTURE=1), behave same as --list
    if do_list_view:
        print_list(aliases, sources)
        return

    # Handle --find / -f (CLI-03)
    if find_pattern is not None:
        matched = [a for a in aliases if find_pattern.lower() in a.name.lower()
                   or find_pattern.lower() in a.template.lower()]
        if not matched:
            click.echo(f"No aliases matching '{find_pattern}'")
        else:
            click.echo(f"{'别名':<20} {'命令'}")
            for a in matched:
                click.echo(f"  {a.name:<20} {a.template}")
        return

    # Alias expansion
    if not alias_input:
        return

    # Build full input text
    input_text = alias_input
    if args:
        input_text = f"{alias_input} {' '.join(args)}"

    # Build matcher
    matcher = AliasMatcher()
    for entry in aliases:
        matcher.add_alias(entry)

    # Tokenize input
    input_tokens = get_tokens(input_text)
    if not input_tokens:
        result = sp.run(input_text, shell=True)
        sys.exit(result.returncode)

    # Find best match
    match_result = matcher.find_best_match(input_tokens)

    if match_result is None:
        # MATCH-07: passthrough for unknown aliases
        result = sp.run(input_text, shell=True)
        sys.exit(result.returncode)

    matched_alias, template, captures, rest_capture, args_start = match_result

    # Extract runtime args
    runtime_args: list = []
    if args_start < len(input_tokens):
        runtime_args = list(input_tokens[args_start:])

    # Expand template
    final_cmd, _ = expand_template(template, captures, rest_capture, runtime_args)

    # Print alias hit message to stderr (per shell L1056)
    entry_name = os.environ.get('WSHA_ENTRY', 'wsha')
    print_alias_hit(entry_name, input_text, final_cmd)

    # Execute and propagate exit code
    exit_code = invoke_cmd(final_cmd)
    sys.exit(exit_code)


if __name__ == '__main__':
    main()
```
