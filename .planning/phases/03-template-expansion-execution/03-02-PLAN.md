**main() function:**
```python
@click.command()
@click.argument('alias_input', required=False)
@click.argument('args', nargs=-1, type=click.UNPROCESSED)
@click.option('--help', '-h', is_flag=True, help='Show help')
def main(alias_input: str, args: tuple, help: bool):
    """wsha - alias command launcher (Python implementation)"""
    if help or not alias_input:
        click.echo("wsha - alias command launcher")
        click.echo("Usage: w <alias> [args...]")
        return
    
    # Build full input text
    input_text = alias_input
    if args:
        input_text = f"{alias_input} {' '.join(args)}"
    
    # Load config
    aliases, errors, sources = load_config()
    
    # Build matcher
    matcher = AliasMatcher()
    for entry in aliases:
        matcher.add_alias(entry)
    
    # Check for passthrough (no alias matched)
    from py.wsha.matching import get_tokens
    input_tokens = get_tokens(input_text)
    if not input_tokens:
        # Passthrough
        import subprocess
        result = subprocess.run(input_text, shell=True)
        raise SystemExit(result.returncode)
    
    # Find best match
    match_result = matcher.find_best_match(input_tokens)
    
    if match_result is None:
        # Passthrough: no alias matched
        import subprocess
        result = subprocess.run(input_text, shell=True)
        raise SystemExit(result.returncode)
    
    matched_alias, template, captures, rest_capture, args_start = match_result
    
    # Get runtime args from input_tokens[args_start:]
    runtime_args = list(input_tokens[args_start:]) if args_start < len(input_tokens) else []
    
    # Expand template
    final_cmd, _ = expand_template(template, captures, rest_capture, runtime_args)
    
    # Invoke command
    exit_code = invoke_cmd(final_cmd)
    raise SystemExit(exit_code)
```

This wires:
1. load_config -> AliasEntry list
2. AliasMatcher.add_alias for each entry
3. get_tokens to parse input_text
4. AliasMatcher.find_best_match -> match_result
5. expand_template with captures, rest_capture, runtime_args
6. invoke_cmd to execute and return exit code
